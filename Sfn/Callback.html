def run_action(msg)
  ui.info("#{msg}... ", :nonewline)
  begin
    result = yield
    ui.puts ui.color('complete!', :green, :bold)
    result
  rescue => e
    ui.puts ui.color('error!', :red, :bold)
    ui.error "Reason - #{e}"
    raise
  end
end
