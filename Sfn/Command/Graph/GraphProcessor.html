def apply_function(hash, funcs=[])
  k, v = hash.first
  if(hash.size == 1)
    case k
    when 'Ref'
      parameters.key?(v) ? parameters[v] : hash
    when 'Fn::Join'
      v.last
    when 'Fn::Select'
      v.last[v.first.to_i]
    else
      hash
    end
  else
    hash
  end
end
