def generate_plan(template, parameters)
  parameters = Smash[parameters.map{|k,v| [k, v.to_s]}]
  result = Smash.new(
    :stacks => Smash.new(
      origin_stack.name => plan_stack(
        origin_stack,
        template,
        parameters
      )
    ),
    :added => Smash.new,
    :removed => Smash.new,
    :replace => Smash.new,
    :interrupt => Smash.new,
    :unavailable => Smash.new,
    :unknown => Smash.new
  )
  result
end
