def scrub!(template)
  parameters = template['Parameters']
  if(parameters)
    parameters.each do |name, options|
      options.delete_if do |attribute, value|
        !ALLOWED_PARAMETER_ATTRIBUTES.include?(attribute)
      end
    end
    template['Parameters'] = parameters
  end
  template
end
